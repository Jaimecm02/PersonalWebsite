import torch.nn as nn

# Define a neural network

    def __init__(self):

            nn.Linear(784, 128),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(128, 10),
            nn.Softmax(dim=1)
        )

    def forward(self, x):
